package com.example.easyboat.model;

public class Club {

    private String nom;
    private String mail;
    private String tel;

    public Club(String nom, String mail, String tel) {
        this.nom = nom;
        this.mail = mail;
        this.tel = tel;
    }

    public String getNom() {
        return "Nom du club : " + nom;
    }

    public void setNom(String nom) {
        this.nom = nom;
    }

    public String getMail() {
        return "Adresse mail du club : " + mail;
    }

    public void setMail(String mail) {
        this.mail = mail;
    }

    public String getTel() {
        return "Numéro de téléphone du club : " + tel;
    }

    public void setTel(String tel) {
        this.tel = tel;
    }
}
